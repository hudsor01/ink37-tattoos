    **Part A: Resource preloading in public layout**

    In src/app/(public)/layout.tsx, add React 19 resource preloading calls at the top of the component function body (before the return statement):

    ```typescript
    import { prefetchDNS, preconnect } from 'react-dom';

    export default function PublicLayout({ children }: { children: React.ReactNode }) {
      // React 19 resource preloading for external services
      prefetchDNS('https://app.cal.com');
      prefetchDNS('https://api.cal.com');
      preconnect('https://fonts.gstatic.com');

      // ... existing layout return
    }
    ```

    These calls are deduplicated by React -- calling them multiple times is safe. They pre-resolve DNS and pre-connect to external domains before the user navigates to pages that use them (Cal.com booking, Google Fonts).

    **Part B: useFormStatus SubmitButton component**

    Create src/components/ui/submit-button.tsx:

    ```typescript
    'use client';
    import { useFormStatus } from 'react-dom';
    import { Button } from '@/components/ui/button';
    import { Loader2 } from 'lucide-react';

    interface SubmitButtonProps extends React.ComponentProps<typeof Button> {
      children: React.ReactNode;
      pendingText?: string;
    }

    export function SubmitButton({ children, pendingText, ...props }: SubmitButtonProps) {
      const { pending } = useFormStatus();
      return (
        <Button type="submit" disabled={pending} {...props}>
          {pending && <Loader2 className="mr-2 h-4 w-4 animate-spin" />}
          {pending ? (pendingText ?? 'Saving...') : children}
        </Button>
      );
    }
    ```

    This component must be rendered INSIDE a `<form>` element -- useFormStatus reads the form's submission state from the nearest parent form. It replaces manual isPending tracking for forms using native form actions.
