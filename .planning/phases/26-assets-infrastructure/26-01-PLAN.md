    **Service worker registration component:**
    Create `src/components/service-worker-registration.tsx` as a client component that registers the service worker on mount:

    ```typescript
    'use client';
    import { useEffect } from 'react';

    export function ServiceWorkerRegistration() {
      useEffect(() => {
        if ('serviceWorker' in navigator) {
          navigator.serviceWorker.register('/sw.js', {
            scope: '/',
            updateViaCache: 'none',
          });
        }
      }, []);
      return null;
    }
    ```
