1. **Create `src/app/api/webhooks/stripe/route.ts`** -- Per D-11, D-12, D-13, D-14, D-15, SEC-05:

```typescript
import { NextResponse } from 'next/server';
import { stripe, stripeCentsToDollars } from '@/lib/stripe';
import { db } from '@/lib/db';
import type Stripe from 'stripe';

const webhookSecret = process.env.STRIPE_WEBHOOK_SECRET!;

export async function POST(request: Request) {
  // D-13 / SEC-05: Raw body for signature verification
  const body = await request.text();
  const sig = request.headers.get('stripe-signature');

  if (!sig) {
    return NextResponse.json({ error: 'Missing stripe-signature header' }, { status: 400 });
  }

  let event: Stripe.Event;
  try {
    event = stripe.webhooks.constructEvent(body, sig, webhookSecret);
  } catch (err) {
    console.error('Webhook signature verification failed:', err);
    return NextResponse.json({ error: 'Invalid signature' }, { status: 400 });
  }

  // D-12: Idempotency -- check if event already processed
  const existingEvent = await db.stripeEvent.findUnique({
    where: { stripeEventId: event.id },
  });
  if (existingEvent) {
    return NextResponse.json({ received: true });
  }

  try {
    // D-14: Handle specified event types
    switch (event.type) {
      case 'checkout.session.completed':
        await handleCheckoutCompleted(event.data.object as Stripe.Checkout.Session);
        break;
      case 'payment_intent.succeeded':
        await handlePaymentSucceeded(event.data.object as Stripe.PaymentIntent);
        break;
      case 'payment_intent.payment_failed':
        await handlePaymentFailed(event.data.object as Stripe.PaymentIntent);
        break;
      case 'charge.refunded':
        await handleChargeRefunded(event.data.object as Stripe.Charge);
        break;
    }

    // Mark event as processed (D-12)
    await db.stripeEvent.create({
      data: {
        stripeEventId: event.id,
        type: event.type,
        processedAt: new Date(),
      },
    });
  } catch (err) {
    console.error(`Webhook handler error for ${event.type}:`, err);
    return NextResponse.json({ error: 'Handler failed' }, { status: 500 });
  }

  return NextResponse.json({ received: true });
}
```

Handler functions (all inside the same file):

a. **`handleCheckoutCompleted(session: Stripe.Checkout.Session)`** -- Per D-15:
   - Extract `tattooSessionId` from `session.metadata`
   - Retrieve PaymentIntent: `stripe.paymentIntents.retrieve(session.payment_intent as string)`
   - Retrieve Charge for receipt URL: `stripe.charges.retrieve(paymentIntent.latest_charge as string)` -- Per D-16, PAY-05
   - Store `charge.receipt_url` on Payment record
   - Calculate `amountInDollars = stripeCentsToDollars(session.amount_total ?? 0)`
   - Atomic transaction (`db.$transaction`):
     - `tx.payment.updateMany({ where: { stripeCheckoutSessionId: session.id }, data: { status: 'COMPLETED', stripePaymentIntentId, receiptUrl, completedAt: new Date() } })`
     - `tx.tattooSession.update({ where: { id: tattooSessionId }, data: { paidAmount: { increment: amountInDollars } } })` -- Per D-06, D-15

b. **`handlePaymentSucceeded(paymentIntent: Stripe.PaymentIntent)`** -- Update any Payment with matching stripePaymentIntentId to COMPLETED (if not already done by checkout.session.completed). This is a safety net. Skip if no matching Payment found.

c. **`handlePaymentFailed(paymentIntent: Stripe.PaymentIntent)`** -- Update Payment with matching stripeCheckoutSessionId (from metadata) to FAILED status.

d. **`handleChargeRefunded(charge: Stripe.Charge)`** -- Update Payment with matching stripePaymentIntentId to REFUNDED. If the charge has a `payment_intent`, find Payment by that ID. Decrement TattooSession.paidAmount by the refunded amount (`stripeCentsToDollars(charge.amount_refunded)`), atomically in a transaction.
