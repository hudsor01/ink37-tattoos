The complete CSP directives (per D-11, D-12, D-13, and research discretion on Google Fonts):

```
default-src 'self';
script-src 'self' 'nonce-${nonce}' 'strict-dynamic' https://app.cal.com${isDev ? " 'unsafe-eval'" : ''};
style-src 'self' 'nonce-${nonce}' 'unsafe-inline';
```

**CRITICAL: style-src includes BOTH 'nonce-${nonce}' AND 'unsafe-inline'.**
This is NOT contradictory -- it addresses the Recharts inline style attribute problem (HIGH review concern):
- Recharts applies inline `style=""` attributes on SVG elements (`<rect style="fill: hsl(...)">`)
- The chart.tsx component also renders `<style dangerouslySetInnerHTML>` for color variables
- CSP Level 3 browsers: nonce controls `<style>` elements, `'unsafe-inline'` is ignored for `<style>` but allows inline style attributes
- CSP Level 2 browsers: `'unsafe-inline'` takes effect since they don't understand nonces, allowing both
- This is the recommended approach from the W3C CSP spec for gradual migration

The nonce still provides value because:
- `<script>` tags require the nonce (no `'unsafe-inline'` in script-src)
- In CSP Level 3 browsers, injected `<style>` tags without nonces are blocked
- Only inline style *attributes* (like Recharts uses) get the `'unsafe-inline'` pass

**Stripe CSP domains NOT added** -- verified that this project uses server-side Stripe only (no `@stripe/stripe-js`, no `loadStripe`, no Stripe Elements). The proxy function structure:
```typescript
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function proxy(request: NextRequest) {
  const { pathname } = request.nextUrl;

  // Skip nonce/CSP for API routes -- they return JSON, not HTML
  // Addresses review concern: unnecessary CSP on API routes adds latency
  if (pathname.startsWith('/api/')) {
    // Still handle the request normally, just no CSP overhead
    return NextResponse.next();
  }

  const nonce = Buffer.from(crypto.randomUUID()).toString('base64');
  const isDev = process.env.NODE_ENV === 'development';

  // Build CSP header with per-request nonce
  const cspHeader = [
    "default-src 'self'",
    `script-src 'self' 'nonce-${nonce}' 'strict-dynamic' https://app.cal.com${isDev ? " 'unsafe-eval'" : ''}`,
    `style-src 'self' 'nonce-${nonce}' 'unsafe-inline'`,
    "font-src 'self'",
    "img-src 'self' data: blob: https:",
    "frame-src 'self' https://app.cal.com",
    "connect-src 'self' https://api.cal.com https://*.sentry.io https://*.ingest.sentry.io",
    "object-src 'none'",
    "base-uri 'self'",
    "form-action 'self'",
    "frame-ancestors 'none'",
    "upgrade-insecure-requests",
  ].join('; ');

  // Auth redirect logic (PRESERVED FROM CURRENT proxy.ts)
  const sessionToken = request.cookies.get('better-auth.session_token');

  if ((pathname.startsWith('/dashboard') || pathname.startsWith('/portal')) && !sessionToken) {
    const response = NextResponse.redirect(new URL('/login', request.url));
    response.headers.set('Content-Security-Policy', cspHeader);
    return response;
  }

  if ((pathname === '/login' || pathname === '/register') && sessionToken) {
    const response = NextResponse.redirect(new URL('/portal', request.url));
    response.headers.set('Content-Security-Policy', cspHeader);
    return response;
  }

  // Normal request -- inject nonce header and CSP
  const requestHeaders = new Headers(request.headers);
  requestHeaders.set('x-nonce', nonce);
  requestHeaders.set('Content-Security-Policy', cspHeader);

  const response = NextResponse.next({ request: { headers: requestHeaders } });
  response.headers.set('Content-Security-Policy', cspHeader);
  return response;
}

export const config = {
  matcher: [
    /*
     * Match all request paths except:
     * - _next/static (static files)
     * - _next/image (image optimization)
     * - favicon.ico, sitemap.xml, robots.txt (static metadata files)
     * - Public assets (images, videos, icons)
     */
    '/((?!_next/static|_next/image|favicon\\.ico|sitemap\\.xml|robots\\.txt|images/|videos/|icons/).*)',
  ],
};
```
