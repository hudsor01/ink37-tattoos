**1. REPLACE src/lib/security/rate-limiter.ts** entirely with Upstash-backed implementation (per D-06, D-07, D-08):

```typescript
import { Ratelimit } from '@upstash/ratelimit';
import { Redis } from '@upstash/redis';

// Graceful degradation: use in-memory for development when Upstash is not configured
function createRedisOrFallback() {
  const url = process.env.UPSTASH_REDIS_REST_URL;
  const token = process.env.UPSTASH_REDIS_REST_TOKEN;
  if (url && token) {
    return new Redis({ url, token });
  }
  // Development fallback -- not suitable for production
  console.warn('[Rate Limiter] UPSTASH_REDIS_REST_URL not set, using ephemeral in-memory storage');
  return undefined;
}

const redis = createRedisOrFallback();

function createLimiter(requests: number, window: string, prefix: string) {
  if (redis) {
    return new Ratelimit({
      redis,
      limiter: Ratelimit.slidingWindow(requests, window as Parameters<typeof Ratelimit.slidingWindow>[1]),
      prefix,
    });
  }
  // In-memory fallback for dev
  return new Ratelimit({
    limiter: Ratelimit.slidingWindow(requests, window as Parameters<typeof Ratelimit.slidingWindow>[1]),
    prefix,
    ephemeralCache: new Map(),
  });
}

// D-07: Per-route rate limiters with separate thresholds
export const rateLimiters = {
  contact: createLimiter(5, '1 m', 'rl:contact'),
  storeDownload: createLimiter(20, '1 m', 'rl:download'),
  portalBilling: createLimiter(10, '1 m', 'rl:billing'),
  webhook: createLimiter(100, '1 m', 'rl:webhook'),
} as const;

// Helper to extract client IP from Request object (API routes)
export function getRequestIp(request: Request): string {
  const xff = request.headers.get('x-forwarded-for');
  if (xff) return xff.split(',')[0].trim();
  return request.headers.get('x-real-ip') ?? '127.0.0.1';
}

// Helper to extract client IP from headers() (server actions)
export function getHeaderIp(hdrs: Headers): string {
  const xff = hdrs.get('x-forwarded-for');
  if (xff) return xff.split(',')[0].trim();
  return hdrs.get('x-real-ip') ?? '127.0.0.1';
}

// D-08: Create 429 response with Retry-After header
export function rateLimitResponse(reset: number) {
  const retryAfter = Math.ceil((reset - Date.now()) / 1000);
  return Response.json(
    { error: 'Too many requests' },
    {
      status: 429,
      headers: {
        'Retry-After': String(Math.max(retryAfter, 1)),
        'X-RateLimit-Reset': String(reset),
      },
    },
  );
}
```

NOTE: `@upstash/ratelimit` v2.x supports creating a Ratelimit without redis (ephemeral mode) -- check the API. If ephemeral mode constructor differs, use the appropriate factory. The key contract: `rateLimiters.X.limit(identifier)` returns `Promise<{ success: boolean; reset: number; limit: number; remaining: number }>`.

**2. CREATE src/lib/security/sanitize.ts** (per D-13, D-15):

```typescript
/**
 * XSS detection for Zod refinements.
 * Rejects strings containing HTML tags, script injections, or event handlers.
 * Used at the validation layer to prevent HTML/script content from being stored.
 */
const HTML_PATTERN = /<[^>]*>|&lt;|&gt;|javascript:|on\w+\s*=/i;

export function noHtml(value: string): boolean {
  return !HTML_PATTERN.test(value);
}

export const noHtmlMessage = 'Text must not contain HTML tags or script content';
```
